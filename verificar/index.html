function doGet(e) { const idBusca = (e.parameter.id || "").trim().toLowerCase(); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; const data = sheet.getDataRange().getValues(); // Buscamos en la Columna AF (Índice 31) for (var i = 1; i < data.length; i++) { // Limpiamos el ID de la celda para comparar let idCelda = String(data[i][31]).trim().toLowerCase(); if (idCelda === idBusca && idBusca !== "") { const respuesta = { status: "valido", nombre: data[i][4], // Columna E evento: "Masterclass Vende Tus Redes", ciudad: data[i][27], // Columna AB fecha: data[i][28] // Columna AC }; return ContentService.createTextOutput(JSON.stringify(respuesta)) .setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({status: "no_encontrado"})) .setMimeType(ContentService.MimeType.JSON); }
